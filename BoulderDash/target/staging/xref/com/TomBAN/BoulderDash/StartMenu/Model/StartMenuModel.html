3   import java.util.ArrayList;
4   import java.util.Observable;
5   
6   import com.TomBAN.BoulderDash.Game.Model.Controllable;
7   import com.TomBAN.BoulderDash.Game.Model.MovementOrder;
8   
9   public class StartMenuModel extends Observable implements Controllable {
10  	ArrayList<LeftRightSelector> selectors;
11  
12  	public StartMenuModel() {
13  		selectors = new ArrayList<LeftRightSelector>();
14  		// TODO Auto-generated constructor stub
15  	}
16  
17  	public void addSelector(LeftRightSelector<?> a) {
18  		selectors.add(a);
19  	}
20  
21  	@Override
22  	public void executeOrder(MovementOrder order) {
23  		switch (order) {
24  		case GoLeft:
25  			selectors.get(selectors.size() - 1).left();
26  			break;
27  		case GoRight:
28  			selectors.get(selectors.size() - 1).right();
29  			break;
30  		case GoUp:
31  			if(selectors.size()>1) {
32  				selectors.remove(selectors.size() - 1);
33  			}
34  		default:
35  			break;
36  		}
37  		notifyObservers();
38  
39  	}
40  
41  	@Override
42  	public void notifyObservers() {
43  		setChanged();
44  		super.notifyObservers();
45  	}
46  	
47  	public int getSelectorNumber() {
48  		return selectors.size();
49  	}
50  
51  	@SuppressWarnings({ "unchecked", "rawtypes" })
52  	public ArrayList<LeftRightSelector> getSelectors() {
53  		return (ArrayList<LeftRightSelector>) selectors.clone();
54  	}
55  }
