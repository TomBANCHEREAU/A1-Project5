34  	@Override
35  	public void draw(Graphics2D graph, GraphicsObserver observer) {
36  
37  		if (model.getModelStatut() == ModelStatut.StartLevelScreen
38  				|| model.getModelStatut() == ModelStatut.WaitingStart) {
39  			drawLoadingScreen(graph, observer);
40  			return;
41  		}
42  
43  		// TODO Auto-generated method stub
44  		final double time = 30;
45  		Player p = model.getPlayers().get(playerId);
46  		if (currentScale == 0) {
47  			currentScale = observer.getWidth() / (16 * TILE_SIZE);
48  		}
49  		if (p.hasWin()) {
50  			if (timeSinceWin < time) {
51  				timeSinceWin++;
52  			}
53  //			currentScale = Math.min(observer.getWidth() / (24F * TILE_SIZE), observer.getHeight() / (12F * TILE_SIZE));
54  
55  			currentScale = currentScale * (time - timeSinceWin) / time + Math.min(
56  					(observer.getWidth() / (float)((model.getMap().getWidth() + 4) * TILE_SIZE)) * timeSinceWin / time,
57  					(observer.getHeight() / (float)((model.getMap().getHeight() + 4) * TILE_SIZE)) * timeSinceWin / time);
58  			currentCenterX = (int) (currentCenterX * (time - timeSinceWin) / time
59  					+ (model.getMap().getWidth() / 2) * TILE_SIZE * timeSinceWin / time);
60  			currentCenterY = (int) (currentCenterY * (time - timeSinceWin) / time
61  					+ (model.getMap().getHeight() / 2) * TILE_SIZE * timeSinceWin / time);
62  		} else {
63  			timeSinceWin = 0;
64  			currentScale = Math.min(observer.getWidth() / (24F * TILE_SIZE), observer.getHeight() / (12F * TILE_SIZE));
65  			currentCenterX = constrain(currentCenterX,
66  					(int) ((p.getX() + 0.5) * TILE_SIZE - observer.getWidth() / currentScale / 6),
67  					(int) ((p.getX() + 0.5) * TILE_SIZE + observer.getWidth() / currentScale / 6));
68  			currentCenterY = constrain(currentCenterY,
69  					(int) ((p.getY() + 0.5) * TILE_SIZE - observer.getHeight() / currentScale / 6),
70  					(int) ((p.getY() + 0.5) * TILE_SIZE + observer.getHeight() / currentScale / 6));
71  		}
72  
73  		final double scale = currentScale;
74  		final int centerX = (int) (currentCenterX * scale);
75  		final int centerY = (int) (currentCenterY * scale);
76  		drawMap(graph, observer, centerX, centerY, scale);
77  		showGUI(graph, observer);
78  
79  		if ((model.getModelStatut() == ModelStatut.EndMapScreen && model.getStatutAvancement()>50)
80  				|| model.getModelStatut() == ModelStatut.WaitingNextMap) {
81  			drawScoreScreen(graph, observer);
82  		}
83  
84  		return;
85  	}

170 	private String getTimeToString(long time) {
171 		final int sec0 = (int) ((time/1000L)%10);
172 		final int sec1 = (int) ((time/10000L)%6);
173 		final int min0 = (int) ((time/60000L));
174 		return ""+min0+":"+sec1+""+sec0;
175 	}

224 	private void drawMap(Graphics2D graph, GraphicsObserver observer, final int centerX, final int centerY,
225 			final double scale) {
226 
227 		final int originX = (centerX - observer.getWidth() / 2);
228 		final int originY = (centerY - observer.getHeight() / 2);
229 
230 		graph.translate(-originX, -originY);
231 		graph.scale(scale, scale);
232 
233 		final int scaledOriginX = (int) ((centerX - observer.getWidth() / 2) / scale);
234 		final int scaledOriginY = (int) ((centerY - observer.getHeight() / 2) / scale);
235 
236 		final int minX = scaledOriginX / TILE_SIZE - 2;
237 		final int minY = scaledOriginY / TILE_SIZE - 2;
238 
239 		final int maxX = (minX + (int) (observer.getWidth() / scale) / TILE_SIZE + 4);
240 		final int maxY = (minY + (int) (observer.getHeight() / scale) / TILE_SIZE + 4);
241 
242 		for (int x = minX; x < maxX; x++) {
243 			for (int y = minY; y < maxY; y++) {
244 				graph.drawImage(model.getMap().getBackgroundImage(), x * TILE_SIZE, y * TILE_SIZE, observer);
245 			}
246 		}
247 		for (int x = minX; x < maxX; x++) {
248 			for (int y = minY; y < maxY; y++) {
249 				final Block b = model.getMap().getBlockAt(x, y);
250 				if (b != null) {
251 					final float bx = b.getX();
252 					final float by = b.getY();
253 					final Image image = b.getImage();
254 					if (image != null) {
255 						final int Xpos = (int) (bx * TILE_SIZE + TILE_SIZE / 2 - image.getWidth(observer) / 2);
256 						final int Ypos = (int) (by * TILE_SIZE + TILE_SIZE / 2 - image.getHeight(observer) / 2);
257 						graph.drawImage(image, Xpos, Ypos, observer);
258 					} else {
259 						graph.drawRect((int) (bx * TILE_SIZE), (int) (by * TILE_SIZE), TILE_SIZE, TILE_SIZE);
260 					}
261 				}
262 
263 			}
264 		}
265 		graph.scale(1 / scale, 1 / scale);
266 		graph.translate(+originX, +originY);
267 	}

274 	private static int constrain(int val, int min, int max) {
275 		if (val > max) {
276 			return max;
277 		}
278 		if (val < min) {
279 			return min;
280 		}
281 		return val;
282 	}
