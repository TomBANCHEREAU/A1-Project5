71  	public Map toRealMap(ArrayList<ControllableController> controllers) {
72  		final Block[][] blocks = new Block[width][height];
73  		final ArrayList<Player> players = new ArrayList<Player>();
74  		final ArrayList<Network> networks = new ArrayList<Network>();
75  		for (int y = 0; y < height; y++) {
76  			for (int x = 0, i = 0; x < width && i < stringMap[y].length(); x++, i++) {
77  				Network network = null;
78  				if (specialChar.contains(stringMap[y].charAt(i))) {
79  					network = getNetwork(networks,(int) stringMap[y].charAt(i + 1) - 48);
80  				}
81  
82  				switch (stringMap[y].charAt(i)) {
83  				case 'O':
84  					blocks[x][y] = new Rock(x, y);
85  					break;
86  				case 'A':
87  					final Player player = new Player(x, y);
88  					blocks[x][y] = player;
89  					players.add(player);
90  					break;
91  				case 'V':
92  					blocks[x][y] = new Diamond(x, y);
93  					break;
94  				case '#':
95  					blocks[x][y] = new Dirt(x, y);
96  					break;
97  				case '@':
98  					blocks[x][y] = new Boundary(x, y);
99  					break;
100 				case '!':
101 					blocks[x][y] = new HardBoundary(x, y);
102 					break;
103 				case '$':
104 					blocks[x][y] = new Exit(x, y);
105 					break;
106 				case 'L':
107 					blocks[x][y] = new Ennemy(x, y, 'L');
108 					break;
109 				case 'U':
110 					blocks[x][y] = new Ennemy(x, y, 'U');
111 					break;
112 				case 'C':
113 					blocks[x][y] = new Ennemy(x, y, 'C');
114 					break;
115 				case 'D':
116 					blocks[x][y] = new Ennemy(x, y, 'D');
117 					break;
118 				case '_':
119 					blocks[x][y] = new PressurePlate(x, y, network);
120 					break;
121 				case ']':
122 					blocks[x][y] = new Door(x, y, network, false);
123 					break;
124 				case '[':
125 					blocks[x][y] = new Door(x, y, network, true);
126 					break;
127 				default:
128 				case ' ':
129 					blocks[x][y] = null;
130 				}
131 				if (specialChar.contains(stringMap[y].charAt(i))) {
132 					i++;
133 				}
134 			}
135 		}
136 		for (int i = 0; i < controllers.size(); i++) {
137 			controllers.get(i).bindControllable(players.get(i));
138 		}
139 		return new Map(width, height, blocks, players, diamondNeeded, networks);

143 	public void addScore(Score newScore) {
144 		for (int i = 0; i < highScore.length; i++) {
145 			if (highScore[i] == null) {
146 				highScore[i] = newScore;
147 				return;
148 			} else if (newScore.getScore() > highScore[i].getScore()) {
149 				Score tmp = highScore[i];
150 				highScore[i] = newScore;
151 				newScore = tmp;
152 			}
153 		}
154 	}

156 	private Network getNetwork(ArrayList<Network> networks,int id) {
157 		for (Network network : networks) {
158 			if (network.getId() == id) {
159 				return network;
160 			}
161 		}
162 		Network n = new Network(id);
163 		networks.add(n);
164 		return n;
165 	}

183 	private void setStringMap(String[] stringMap) {
184 		if (stringMap.length != height) {
185 			throw new RuntimeException("The specified height does not fit with the String Array (Specified height = "
186 					+ height + " String Array height = " + stringMap.length);
187 		}
188 		int playerCount = 0;
189 		int diamondCount = 0;
190 		boolean haveExit = false;
191 		for (int y = 0; y < height; y++) {
192 			int i, x;
193 			for (x = 0, i = 0; i < stringMap[y].length(); x++, i++) {
194 				if (DiamondChar.contains(stringMap[y].charAt(i))) {
195 					diamondCount++;
196 				}
197 				if (basicChar.contains(stringMap[y].charAt(i))) {
198 					switch (stringMap[y].charAt(i)) {
199 					case 'A':
200 						playerCount++;
201 						break;
202 					case '$':
203 						haveExit = true;
204 						break;
205 
206 					default:
207 						break;
208 					}
209 				} else if (specialChar.contains(stringMap[y].charAt(i))) {
210 					i++;
211 				} else {
212 					throw new RuntimeException("unknown Block : (char:'" + stringMap[y].charAt(i) + "',int: "
213 							+ (int) stringMap[y].charAt(i) + ",hex: " + Integer.toHexString(stringMap[y].charAt(i))
214 							+ ")");
215 				}
216 			}
217 
218 			if (this.width != x) {
219 				throw new RuntimeException("The specified width does not fit with the String length (Specified width = "
220 						+ this.width + " String width = " + x + " at line : " + y + ")");
221 			}
222 		}
223 
224 		if (!haveExit) {
225 			throw new RuntimeException("The map does not have an Exit");
226 		}
227 		if (playerCount != this.playerCount) {
228 			throw new RuntimeException(
229 					"The specified player count does not fit with the player count present on the map (Specified player count = "
230 							+ this.playerCount + " map player count = " + playerCount);
231 		}
232 		if (this.diamondNeeded > diamondCount) {
233 			throw new RuntimeException(
234 					"The specified diamondNeeded does not fit with the diamond count on the map (Specified diamondNeeded = "
235 							+ this.diamondNeeded + " map diamond count = " + diamondCount);
236 		}
237 
238 		this.stringMap = stringMap;
239 	}
