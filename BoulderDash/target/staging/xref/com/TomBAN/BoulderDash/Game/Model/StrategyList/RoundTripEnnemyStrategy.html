9   public class RoundTripEnnemyStrategy extends Strategy<Ennemy> {
10  	public RoundTripEnnemyStrategy(Ennemy block,Direction direction) {
11  		super(block);
12  		block.setDirection(direction);
13  	}
14  
15  	@Override
16  	public void strategy() {
17  		Block goal = block.getBlockAtFront();
18  		if(goal == null) {
19  			block.move(block.getDirection(), 5);
20  		}else if(goal instanceof Player) {
21  			((Player) goal).getKilled();
22  			block.move(block.getDirection(), 5);
23  		}else {
24  			block.setDirection(Direction.values()[(block.getDirection().ordinal()+2)%4]);
25  		}
26  	}
27  
28  }
