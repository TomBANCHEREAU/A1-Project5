11  /**
12   * A basic JFrame implementing the Observer interface
13   * @author Tom BANCHEREAU
14   */
15  public class BoulderDashFrame extends JFrame implements Observer{
16  	private static final long serialVersionUID = -1116924846179363004L;
17  	/**
18  	 * Create a new basic Window 
19  	 * @param width
20  	 * width of the window
21  	 * @param height
22  	 * height of the window
23  	 */
24  	public BoulderDashFrame(int width,int height) {
25  		this.setSize(width, height);
26  		this.setTitle("BoulderDash");
27  		this.setLocationRelativeTo(null);
28  		this.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
29  //		this.setResizable(false);
30  		this.setVisible(true);
31  	}
32  	public void update(Observable o, Object arg) {
33  		repaint();
34  	}
