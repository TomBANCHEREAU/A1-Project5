31  	public void addBlock(Block b) {
32  		if(b!=null) {
33  			blocks[b.getxIndex()][b.getyIndex()] = b;
34  			b.bindMap(this);
35  		}
36  	}

38  	private void bindMapToBlocks() {
39  		for(Block[] blockList : blocks) {
40  			for(Block block : blockList) {
41  				if(block!=null) {
42  					block.bindMap(this);
43  				}
44  			}
45  		}
46  	}

64  	public void updateAllBlock() {
65  		update++;
66  		// for(int y=getHeight()-1;y>=0;y--) {
67  		for (int y = 0; y < getHeight(); y++) {
68  			for (int x = 0; x < getWidth(); x++) {
69  				if (blocks[x][y] != null) {
70  					blocks[x][y].update(update);
71  				}
72  			}
73  		}
74  
75  	}

77  	public void moveBlock(Block b, int x, int y) {
78  		if(getBlockAt(x, y) == null) {
79  			if (x >= 0 && x < width && y >= 0 && y < height) {
80  				blocks[b.getxIndex()][b.getyIndex()] = null;
81  				blocks[x][y] = b;
82  				b.setxIndex(x);
83  				b.setyIndex(y);
84  			}
85  		}
86  
87  	}
88  
89  	public void removeBlock(Block b) {
90  		if (getBlockAt(b.getxIndex(), b.getyIndex()) == b) {
91  			blocks[b.getxIndex()][b.getyIndex()] = null;
92  		}
93  	}

101 	public Block getBlockAt(int x, int y) {
102 		if (x >= 0 && x < width && y >= 0 && y < height) {
103 			return blocks[x][y];
104 		} else {
105 			return new OutOfTheMap(x, y);
106 		}
107 	}

110 	public boolean won() {
111 		for (Player p : getPlayers()) {
112 			if (!p.hasWin()) {
113 				return false;
114 			}
115 		}
116 		return true;
117 	}

119 	public boolean loose() {
120 		for (Player p : getPlayers()) {
121 			if (p.isDead()) {
122 				return true;
123 			}
124 		}
125 		return false;
126 	}
