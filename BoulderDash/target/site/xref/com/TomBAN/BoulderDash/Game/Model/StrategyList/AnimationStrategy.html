6   public class AnimationStrategy extends Strategy<MovableBlock> {
7   	private final Strategy next;
8   	private final int xGoal,yGoal;
9   	private int time;
10  
11  
12  
13  	public AnimationStrategy(MovableBlock block, Strategy next, int xGoal, int yGoal, int time) {
14  		super(block);
15  		this.next = next;
16  		this.xGoal = xGoal;
17  		this.yGoal = yGoal;
18  		this.time = time;
19  	}
20  
21  
22  
23  	@Override
24  	public void strategy() {
25  		// TODO Auto-generated method stub
26  		block.setX(block.getX() + (-block.getX()+xGoal)/time);
27  		block.setY(block.getY() + (-block.getY()+yGoal)/time);
28  		time--;
29  		if(time==0) {
30  			block.setStrategy(next);
31  			block.setX(block.getxIndex());
32  			block.setY(block.getyIndex());
33  			next.strategy();
34  		}
35  	}
36  	public int getTime() {
37  		return time;
38  	}
